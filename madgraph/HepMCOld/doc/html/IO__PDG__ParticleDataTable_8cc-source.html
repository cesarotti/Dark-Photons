00063     void IO_PDG_ParticleDataTable::read_entry( ParticleDataTable* pdt ) {
00064         int position = m_file.tellg();
00065         // variables for input
00066         char code;
00067         int id[4] = { 0,0,0,0 };
00068         int charge[4] = { 999, 999, 999, 999 };
00069         double value;
00070         std::string name;
00071 
00072         code = m_file.get();
00073         if ( code == EOF ) {
00074             m_file.clear( std::ios::eofbit );
00075             return;
00076         }
00077 
00078         if ( code == '\n' ) return; // ignore empty lines.
00079 
00080         if ( code!='M' && code!='W' ) {
00081             std::cerr << "IO_PDG_ParticleDataTable::read_entry cannot "
00082                       << "understand syntax, setting badbit."
00083                       << std::endl;
00084             m_file.putback( code );
00085             m_file.clear( std::ios::badbit ); 
00086             return;
00087         } 
00088         // read the particle ID's -- here I assume there is atleast one
00089         // white space b/t the 4 id columns.
00090         for ( int id_i1 = 0; id_i1 < 4; ++id_i1 ) {
00091             m_file.seekg( position+1+(id_i1*8) );
00092             for ( int i = 0; i < 8; ++i ) {
00093                 if ( !isspace(m_file.peek()) ) {
00094                     //m_file.width(8-i); // has no effect
00095                     // how do I limit to reading only 8-i characters?
00096                     // setw(8-i) has no effect either.
00097                     m_file >> id[id_i1];
00098                     break;
00099                 }
00100                 m_file.ignore(); 
00101             }
00102         }
00103         //m_file.width(0);
00104         m_file.seekg(position+33);
00105         m_file >> value;
00106         m_file.seekg(position+66);
00107         m_file >> name;
00108         // loop over all remaining columns of the name section, where the 
00109         // charge is stored --- until \n is reached
00110         // Here I assume there is atleast one white space between the name
00111         // and the charge. This is true for the 1998 PDG table, and I expect
00112         // it will always be true.
00113         int id_i2 = 0;
00114         while ( m_file.rdstate() == 0 ) {
00115             char c = m_file.get();
00116             if ( c == '\n' || c == EOF ) break;
00117             if ( isspace(c) || c==',' ) {
00118             } else if ( c=='-' ) { 
00119                 charge[id_i2] = -1; 
00120                 ++id_i2; 
00121             } else if ( c=='0' ) {
00122                 charge[id_i2] = 0;
00123                 ++id_i2; 
00124             } else if ( c=='+' ) {
00125                 charge[id_i2] = 1;
00126                 if ( m_file.peek() == '+' ) {
00127                     charge[id_i2] = 2;
00128                     m_file.ignore();
00129                 }
00130                 ++id_i2;
00131             }
00132         }
00133         // add the new information to the particle data table.
00134         for ( int id_i3 = 0; id_i3 < 4; ++id_i3 ) {
00135             if ( id[id_i3] == 0 ) continue;
00136             std::string scharge;
00137             if ( charge[id_i3] == -1 ) { 
00138                 scharge = "-";
00139             } else if ( charge[id_i3] == 1 ) { 
00140                 scharge = "+";
00141             } else if ( charge[id_i3] == 2 ) { 
00142                 scharge = "++"; 
00143             }
00144             ParticleData* pdata = pdt->find( id[id_i3] );
00145             if ( pdata ) {
00146                 if ( code == 'M' ) {
00147                     pdata->set_mass( value );
00148                 } else if ( code == 'W' ) {
00149                     pdata->set_clifetime( clifetime_from_width(value) );
00150                 }
00151             } else if ( code == 'M' ) {
00152                 pdata = new ParticleData( name+scharge, 
00153                                           id[id_i3], (double)charge[id_i3], 
00154                                           value );
00155                 pdt->insert(pdata);
00156             } else if ( code == 'W' ) {
00157                 pdata = new ParticleData( name+scharge, 
00158                                           id[id_i3], (double)charge[id_i3],
00159                                           0, clifetime_from_width(value) );
00160                 pdt->insert(pdata);
00161             }
00162         }
00163     }

00181             // build the particle
00182             ParticleData* pdata = pdt.find(id[i]);
00183             if ( pdata ) {
00184                 mass[i] = pdata->mass();
00185                 pdt.erase(pdata);
00186                 delete pdata;
00187             }
00188             pdata = new ParticleData( name[i], id[i], charge[i], mass[i], -1.,
00189                                       .5 );
00190             pdt.insert(pdata);
00191             // build the antiparticle
00192             pdata = pdt.find(-1*id[i]);
00193             if ( pdata ) {
00194                 pdt.erase(pdata);
00195                 delete pdata;
00196             }
00197             pdata = new ParticleData( name[i]+"~", -1*id[i], -1*charge[i], 
00198                                       mass[i], -1., .5 );
