00005 // generate events with Pythia, write a file, and read the resulting output
00006 // Notice that we use scope to explicitly close the ouput files.
00007 // The two output files should be the same size, but because particles are

00011 // gmake examples/example_MyPythiaRead.exe

00013 // In this example the precision and number of entries for the HEPEVT 
00014 // fortran common block are explicitly defined to correspond to those

00017 // If you get funny output from HEPEVT in your own code, probably you have

00051         //
00052         //........................................EVENT LOOP
00053         for ( int i = 1; i <= 100; i++ ) {
00054             if ( i%50==1 ) std::cout << "Processing Event Number " 
00055                                      << i << std::endl;

00057             // pythia pyhepc routine converts common PYJETS in common HEPEVT

00065             // we also need to delete the created event from memory
00066             delete evt;
00067         }
00068         //........................................TERMINATION
00069         // write out some information from Pythia to the screen
00070         call_pystat( 1 );

00073     //........................................define an input scope

00075         // now read the file we wrote
