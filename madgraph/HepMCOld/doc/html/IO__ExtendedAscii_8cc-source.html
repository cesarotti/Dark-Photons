00678     bool IO_ExtendedAscii::eat_key( std::iostream& in, const char* key ) {
00683         int key_length = strlen(key);
00684         // below is the only way I know of to get a variable length string
00685         //  conforming to ansi standard.
00686         char* c = new char[key_length +1];
00687         int i=0;
00688         // read the stream until get fails (because of EOF), a character
00689         //  doesn't match a character in the string, or all characters in
00690         //  the string have been checked and match.
00691         while ( in.get(c[i]) && c[i]==key[i] && i<key_length ) {
00692             ++i;
00693         }
00694         if ( i == key_length ) {
00695             delete [] c;
00696             return 1;
00697         }
00698         //
00699         // if we get here, then we have eaten the wrong this and we must put it
00700         // back
00701         while ( i>=0 ) in.putback(c[i--]); 
00702         delete c;
00703         return 0;
00704     }
